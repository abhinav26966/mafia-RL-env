# Core training libs — only the conflict-prone ones are hard-pinned.
# Loosen these only after testing; transformers 5.x and pydantic 2.12 break
# Unsloth + TRL 0.20 in confusing ways.
trl==0.20.0
datasets>=2.18,<3.0
accelerate>=1.0,<2.0          # mergekit needs >=1.1; pip resolves at install
peft>=0.12,<0.14

# mergekit is imported by trl.trainer.grpo_trainer (LoRA adapter merging path).
# Recent mergekit versions are incompatible with pydantic >= 2.12 (raw
# torch.Tensor fields fail Pydantic v2 schema generation). We stub mergekit
# in training/train_grpo.py at import time, so we don't actually need the
# package — but installing an old version doesn't hurt as a fallback.
mergekit
# Pin pydantic explicitly — 2.12+ breaks mergekit's torch.Tensor schemas,
# 2.11 breaks some internal pydantic calls in our path. 2.10.6 is verified.
pydantic==2.10.6
